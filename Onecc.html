            fileUpload.addEventListener('change', function(e) {

            // CSV parsing function

            processFileBtn.addEventListener('click', function() {

                const file = fileUpload.files[0];

                if (!file) return;

                

                const reader = new FileReader();

                reader.onload = function(e) {

                    try {

                        const csvData = e.target.result;

                        const nameColumn = document.getElementById('name-column').value || 'name';

                        const phoneColumn = document.getElementById('phone-column').value || 'phone';

                        

                        // Parse CSV

                        const lines = csvData.split('\n');

                        const headers = lines[0].split(',').map(h => h.trim());

                        

                        const nameIndex = headers.findIndex(h => h.toLowerCase() === nameColumn.toLowerCase());

                        const phoneIndex = headers.findIndex(h => h.toLowerCase() === phoneColumn.toLowerCase());

                        

                        if (nameIndex === -1 || phoneIndex === -1) {

                            alert(`Column name "${nameColumn}" or "${phoneColumn}" not found in CSV headers.`);

                            return;

                        }

                        

                        callListData = [];

                        for (let i = 1; i < lines.length; i++) {

                            if (!lines[i].trim()) continue;

                            

                            const columns = lines[i].split(',').map(col => col.trim());

                            const contact = {

                                id: `contact-${i}`,

                                name: columns[nameIndex],

                                phone: columns[phoneIndex],

                                callStatus: 'Not Called',

                                lastCalled: null

                            };

                            

                            // Add additional columns as hidden data

                            headers.forEach((header, idx) => {

                                if (idx !== nameIndex && idx !== phoneIndex) {

                                    contact[header] = columns[idx];

                                }

                            });

                            

                            // Check call history for this contact

                            const historyEntry = callHistory.find(h => h.phone === contact.phone);

                            if (historyEntry) {

                                contact.callStatus = 'Called';

                                contact.lastCalled = historyEntry.timestamp;

                                contact.notes = historyEntry.notes;

                            }

                            

                            callListData.push(contact);

                        }

                        

                        rowCount.textContent = `${callListData.length} contacts`;

                        document.getElementById('stat-total-contacts').textContent = callListData.length;

                        

                        // Show the call list

                        document.getElementById('call-list-empty').classList.add('hidden');

                        document.getElementById('call-list-container').classList.remove('hidden');

                        

                        // Update the display

                        filteredData = [...callListData];

                        renderCallList();

                        updateCallStats();

                        renderCallHistory();

                        

                    } catch (error) {

                        console.error('Error parsing CSV:', error);

                        alert('Error parsing CSV file. Please check the format and try again.');

                    }

                };

                reader.readAsText(file);

            });

            

            // Call list rendering

            function renderCallList() {

                const tableBody = document.getElementById('call-list-body');

                tableBody.innerHTML = '';

                

                const startIndex = (currentPage - 1) * rowsPerPage;

                const endIndex = Math.min(startIndex + rowsPerPage, filteredData.length);

                const pageData = filteredData.slice(startIndex, endIndex);

                

                if (pageData.length === 0) {

                    const row = document.createElement('tr');

                    row.innerHTML = `

                        <td colspan="4" class="px-6 py-4 text-center text-gray-500">

                            No matching contacts found

                        </td>

                    `;

                    tableBody.appendChild(row);

                } else {

                    pageData.forEach(contact => {

                        const row = document.createElement('tr');

                        row.className = 'hover:bg-gray-50';

                        

                        const lastCalledText = contact.lastCalled ? 

                            new Date(contact.lastCalled).toLocaleString('en-US', {

                                month: 'short', 

                                day: 'numeric',

                                year: 'numeric',

                                hour: 'numeric',

                                minute: '2-digit'

                            }) : 'Never';

                        

                        const statusColor = contact.callStatus === 'Called' ? 'bg-green-100 text-green-800' : 'bg-yellow-100 text-yellow-800';

                        

                        row.innerHTML = `

                            <td class="px-6 py-4 whitespace-nowrap">

                                <div class="text-sm font-medium text-gray-900">${contact.name}</div>

                            </td>

                            <td class="px-6 py-4 whitespace-nowrap">

                                <span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full ${statusColor}">

                                    ${contact.callStatus}

                                </span>

                            </td>

                            <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">

                                ${lastCalledText}

                            </td>

                            <td class="px-6 py-4 whitespace-nowrap text-right text-sm font-medium">

                                <button class="text-blue-600 hover:text-blue-900 call-button" data-id="${contact.id}">

                                    <i class="fas fa-phone"></i> Call

                                </button>

                            </td>

                        `;

                        

                        tableBody.appendChild(row);

                    });

                }

                

                // Update pagination

                document.getElementById('showing-count').textContent = pageData.length;

                document.getElementById('total-count').textContent = filteredData.length;

                document.getElementById('current-page').textContent = currentPage;

                document.getElementById('next-page').disabled = endIndex >= filteredData.length;

                

                // Add event listeners to call buttons

                document.querySelectorAll('.call-button').forEach(button => {

                    button.addEventListener('click', function() {

                        const contactId = this.getAttribute('data-id');

                        const contact = callListData.find(c => c.id === contactId);

                        if (contact) {

                            initiateCall(contact);

                        }

                    });

                });

            }

            

            // Call initiation

            function initiateCall(contact) {

                currentContact = contact;

                document.getElementById('calling-name').textContent = contact.name;

                document.getElementById('call-notes').value = contact.notes || '';

                document.getElementById('call-modal').classList.remove('hidden');

                

                // In a real app, this would integrate with a telephony API

                console.log(`Initiating call to ${contact.name} at ${contact.phone}`);

            }

            

            // Call modal handling

            document.getElementById('close-call-modal').addEventListener('click', function() {

                document.getElementById('call-modal').classList.add('hidden');

                currentContact = null;

            });

            

            document.getElementById('end-call').addEventListener('click', function() {

                endCallAndSave();

            });

            

            document.getElementById('save-call').addEventListener('click', function() {

                endCallAndSave();

            });

            

            function endCallAndSave() {

                if (!currentContact) return;

                

                const notes = document.getElementById('call-notes').value;

                const timestamp = new Date().toISOString();

                

                // Update contact status

                const contact = callListData.find(c => c.id === currentContact.id);

                if (contact) {

                    contact.callStatus = 'Called';

                    contact.lastCalled = timestamp;

                    contact.notes = notes;

                }

                

                // Add to call history

                const historyEntry = {

                    id: `history-${Date.now()}`,

                    name: currentContact.name,

                    phone: currentContact.phone,

                    timestamp: timestamp,

                    notes: notes

                };

                

                // Remove any previous entries with the same phone number

                callHistory = callHistory.filter(h => h.phone !== currentContact.phone);

                

                // Add new entry at the beginning

                callHistory.unshift(historyEntry);

                

                // Save to localStorage

                localStorage.setItem('callHistory', JSON.stringify(callHistory));

                

                // Update UI

                document.getElementById('call-modal').classList.add('hidden');

                renderCallList();

                updateCallStats();

                renderCallHistory();

                currentContact = null;

            }

            

            // Search and filtering

            document.getElementById('search-input').addEventListener('input', function() {

                const searchTerm = this.value.toLowerCase();

                

                if (searchTerm.trim() === '') {

                    filteredData = [...callListData];

                } else {

                    filteredData = callListData.filter(contact => 

                        contact.name.toLowerCase().includes(searchTerm)

                    );

                }

                

                currentPage = 1;

                renderCallList();

            });

            

            // Sorting

            document.getElementById('sort-select').addEventListener('change', function() {

                const sortOption = this.value;

                

                switch (sortOption) {

                    case 'name-asc':

                        filteredData.sort((a, b) => a.name.localeCompare(b.name));

                        break;

                    case 'name-desc':

                        filteredData.sort((a, b) => b.name.localeCompare(a.name));

                        break;

                    case 'recent':

                        filteredData.sort((a, b) => {

                            if (!a.lastCalled) return 1;

                            if (!b.lastCalled) return -1;

                            return new Date(b.lastCalled) - new Date(a.lastCalled);

                        });

                        break;

                    case 'not-called':

                        filteredData.sort((a, b) => {

                            if (a.callStatus === 'Not Called' && b.callStatus !== 'Not Called') return -1;

                            if (a.callStatus !== 'Not Called' && b.callStatus === 'Not Called') return 1;

                            return a.name.localeCompare(b.name);

                        });

                        break;

                }

                

                currentPage = 1;

                renderCallList();

            });

            

            // Pagination

            document.getElementById('prev-page').addEventListener('click', function() {

                if (currentPage > 1) {

                    currentPage--;

                    renderCallList();

                }

            });

            

            document.getElementById('next-page').addEventListener('click', function() {

                if ((currentPage * rowsPerPage) < filteredData.length) {

                    currentPage++;

                    renderCallList();

                }

            });

            

            // Stats and reporting

            function updateCallStats() {

                const totalContacts = callListData.length;

                const callsMade = callListData.filter(c => c.callStatus === 'Called').length;

                const completion = totalContacts > 0 ? Math.round((callsMade / totalContacts) * 100) : 0;

                

                document.getElementById('stat-total-contacts').textContent = totalContacts;

                document.getElementById('stat-calls-made').textContent = callsMade;

                document.getElementById('stat-completion').textContent = `${completion}%`;

            }

            

            function renderCallHistory() {

                const historyBody = document.getElementById('call-history-body');

                historyBody.innerHTML = '';

                

                if (callHistory.length === 0) {

                    const row = document.createElement('tr');

                    row.innerHTML = `

                        <td colspan="3" class="px-6 py-4 text-center text-gray-500">

                            No call history yet

                        </td>

                    `;

                    historyBody.appendChild(row);

                } else {

                    // Show only the 5 most recent calls

                    const recentHistory = callHistory.slice(0, 5);

                    

                    recentHistory.forEach(entry => {

                        const row = document.createElement('tr');

                        const callDate = new Date(entry.timestamp).toLocaleString('en-US', {

                            month: 'short', 

                            day: 'numeric',

                            year: 'numeric',

                            hour: 'numeric',

                            minute: '2-digit'

                        });

                        

                        row.innerHTML = `

                            <td class="px-6 py-4 whitespace-nowrap">

                                <div class="text-sm font-medium text-gray-900">${entry.name}</div>

                            </td>

                            <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">

                                ${callDate}

                            </td>

                            <td class="px-6 py-4 text-sm text-gray-500">

                                ${entry.notes || 'No notes'}

                            </td>

                        `;

                        

                        historyBody.appendChild(row);

                    });

                }

            }
